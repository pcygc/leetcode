class Solution {
    public int lengthOfLongestSubstring(String s) {
        //count用来记录最长的子串长度
        int count = 0;
        //list记录不重复的子串
        List<Object> list = new ArrayList<>();
        int length = s.length();
        for (int i = 0;i<length;i++){
            char c = s.charAt(i);
            //如果c在list中存在去除掉和c相同的及前面的字符保证list中无重复的字符
            while (list.indexOf(c)>=0) {
                list.remove(0);
            }
            //list中追加c
            list.add(c);
            //list的长度大于count则替换
            count = list.size() > count ? list.size() : count;
        }
        return count;
    }
}
